1. First point → win correlation
We match the first_pt column (who scored first) against which team they were on, then compare that to the winner column. The key trick is parsing the comma-separated team strings into a list before checking membership.
fp = games[games['first_pt'].notna()].copy() def get_members(s): return [x.strip() for x in str(s).split(',')] # Which team did the first scorer belong to? fp['first_scorer_team'] = fp.apply(lambda r: 'Team_1' if r['first_pt'] in get_members(r['team_1']) else 'Team_2', axis=1) # Did that team win? fp['first_scored_won'] = ( fp['first_scorer_team'] == fp['winner'] ).astype(int) print(fp['first_scored_won'].mean()) # overall win rate print(fp[fp['margin']<=2]['first_scored_won'].mean()) # close games only
2. Session game number effect
We sort by date then game_id to get chronological order within each session. groupby(date).cumcount() + 1 assigns each game its position within that day (1st, 2nd, 3rd...). We then check average win% at each position — result was exactly 50% at every slot (null result, but a useful one).
games_sorted = games.sort_values(['date','game_id']).copy() games_sorted['session_game_num'] = ( games_sorted.groupby('date').cumcount() + 1 ) # Merge position onto master sheet master_sess = master.merge( games_sorted[['game_id','session_game_num']], on='game_id', how='left' ) # Average win% at each game position in the session result = master_sess.groupby('session_game_num')['win_bin'].agg( ['mean','count'] ) # Result: 0.5 at every position — no fatigue or warm-up effect
3. Post-blowout-loss performance
For each player, we sort their games chronologically. For each pair of consecutive games, we check if the previous game was a loss AND had a margin ≥10. If so, we record the next game's outcome.
for player in all_players: pdf = (master[master['player']==player] .sort_values('game_id').reset_index(drop=True)) next_win = [] for i in range(1, len(pdf)): prev = pdf.iloc[i-1] if prev['win_bin'] == 0: # previous was a loss gid = prev['game_id'] g = games[games['game_id'] == gid] if len(g) > 0 and g['margin'].values[0] >= 10: curr = pdf.iloc[i] # next game next_win.append(curr['win_bin']) print(f"{player}: {np.mean(next_win):.1%} win rate post-blowout")
4. Contested vs open shooting gap
We filter to only games where contested data exists (att_1pt_cont > 0), then sum total makes and attempts for open and contested separately per player. Dividing gives career-level percentages we can compare directly.
cont_data = master[master['att_1pt_cont'] > 0].copy() for player in all_players: pdf = cont_data[cont_data['player'] == player] open_pct = pdf['made_1pt'].sum() / pdf['att_1pt'].sum() cont_pct = (pdf['made_1pt_cont'].sum() / pdf['att_1pt_cont'].sum()) gap = open_pct - cont_pct print(f"{player}: open={open_pct:.1%} " f"contested={cont_pct:.1%} gap={gap:.1%}")
5. Shooting streakiness (coefficient of variation)
Standard deviation alone can't compare players with different averages — a player shooting 40% will naturally have higher raw variance than one shooting 20%. CV = std / mean normalises this, giving a fair comparison of relative consistency regardless of average.
for player in all_players: pdf = master[master['player'] == player] pct = (pdf['made_1pt'] / pdf['att_1pt'].replace(0, np.nan) ).dropna() cv = pct.std() / pct.mean() print(f"{player}: CV = {cv:.3f}") # Lower CV = more consistent shooter across games # Higher CV = streaky — big variation game to game
6. Opponent-adjusted performance
We build a dictionary keyed by (game_id, player) that stores that player's opponents as a sorted tuple. This lets us merge opponent context onto the master sheet efficiently, then groupby opponents to find win rate patterns.
opp_map = {} for _, row in games.iterrows(): t1 = [p.strip() for p in str(row['team_1']).split(',')] t2 = [p.strip() for p in str(row['team_2']).split(',')] for p in t1: opp_map[(row['game_id'], p)] = tuple(sorted(t2)) for p in t2: opp_map[(row['game_id'], p)] = tuple(sorted(t1)) master['opponents'] = master.apply( lambda r: opp_map.get((r['game_id'], r['player']), None), axis=1 ) # Win rate against each opponent combination opp_stats = (master[master['player']=='E.S'] .groupby('opponents')['win_bin'] .agg(['mean','count']) .query('count >= 2'))
7. Logistic regression — what predicts winning
We standardise all features with StandardScaler so coefficients are comparable regardless of units (points are in double digits, steals in fractions). Then fit LogisticRegression. Positive coefficient = that stat is associated with winning. We sort by absolute value to see what matters most.
from sklearn.linear_model import LogisticRegression from sklearn.preprocessing import StandardScaler feat_cols = ['pts','efg','trb','ass','stl','blk','to'] lr_df = master[feat_cols + ['win_bin']].dropna() # Standardise so coefficients are comparable across stats X = StandardScaler().fit_transform(lr_df[feat_cols]) y = lr_df['win_bin'].values lr = LogisticRegression(max_iter=1000) lr.fit(X, y) # Sort by magnitude — what matters most coefs = sorted(zip(feat_cols, lr.coef_[0]), key=lambda x: abs(x[1]), reverse=True) for feat, coef in coefs: print(f"{feat}: {coef:+.3f}") # Result: ass +1.097, pts +0.836, efg +0.415... # Assists predicts winning more than points
8. K-means player clustering
K-means groups players by minimising within-cluster variance across all stats. We standardise first (same reason as logistic regression — units matter). n_clusters=3 was chosen by inspection — 6 players into more clusters gives meaningless tiny groups.
from sklearn.cluster import KMeans cluster_cols = ['pts','efg','trb','ass','stl','blk','to'] cdf = master.groupby('player')[cluster_cols].mean().dropna() # Standardise before clustering X = StandardScaler().fit_transform(cdf) km = KMeans(n_clusters=3, random_state=42, n_init=10) cdf['cluster'] = km.fit_predict(X) for c in sorted(cdf['cluster'].unique()): members = cdf[cdf['cluster']==c].index.tolist() means = cdf[cdf['cluster']==c][cluster_cols].mean() print(f"Cluster {c}: {members}") print(f" pts={means['pts']:.1f} efg={means['efg']:.3f}")
9. Optimal lineup by current form
We take each player's last 10 games, compute their average eFG% and win%, then create a combined form score. For each possible 2 or 3-player combination we sum their form scores — the combination with the highest total is the current hottest lineup.
from itertools import combinations form = {} for player in all_players: pdf = (master[master['player']==player] .sort_values('game_id').tail(10)) form[player] = { 'efg': pdf['efg'].mean(), 'win_pct': pdf['win_bin'].mean() } # Score every possible 2-player combination best_duo = max( combinations(all_players, 2), key=lambda d: sum( form[p]['efg'] + form[p]['win_pct'] for p in d ) ) print(f"Best 2v2 by form: {best_duo[0]} & {best_duo[1]}") # Same for 3-player best_trio = max( combinations(all_players, 3), key=lambda t: sum( form[p]['efg'] + form[p]['win_pct'] for p in t ) ) print(f"Best 3v3 by form: {', '.join(best_trio)}")
